* = optional parameter

### USERS

# POST /api/1/users
# Create a user account + signin
Request:
  {
    user: {
      email: 'toto@toto.com',
      password: 'password',
      *nickname: 'toto'
    }
  }

Reponse:
  {
    email: 'toto@toto.com',
    nickname: 'toto'
  }
  + header SESSION-TOKEN

# GET /api/1/user
Request:
  {}
  + header SESSION-TOKEN

Reponse:
  {
    email: 'toto@toto.com',
    nickname: 'toto'
  }
  + header SESSION-TOKEN
